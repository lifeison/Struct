Node *first(int d)// Ôîđěčđîâŕíčĺ ďĺđâîăî ýëĺěĺíňŕ
{
	Node *pv = new Node;
	pv->d = d; pv->next = 0; pv->prev = 0;
	return pv;
}

void add(Node **pend, int d)// Äîáŕâëĺíčĺ â ęîíĺö ńďčńęŕ
{
	Node *pv = new Node;
	pv->d = d; pv->next = 0; pv->prev = *pend;
	(*pend)->next = pv;
	*pend = pv;
}
Node *find(Node *const pbeg, int d)// Ďîčńę ýëĺěĺíňŕ ďî ęëţ÷ó
{
	Node *pv = pbeg;
	while (pv)
	{
		if (pv->d == d)break;
		pv = pv->next;
	}
	return pv;
}
Node *insert(Node *const pbeg, Node **pend, int key, int d)// Âńňŕâęŕ ýëĺěĺíňŕ
{
	if (Node *pkey = find(pbeg, key))
	{
		Node *pv = new Node;
		pv->d = d;
		pv->next = pkey->next;// óńňŕíîâëĺíčĺ ńâ˙çč íîâîăî óçëŕ ń ďîńëĺäóţůčě
		pv->prev = pkey;// óńňŕíîâëĺíčĺ ńâ˙çč íîâîăî óçëŕ ń ďđĺäűäóůčě
		pkey->next = pv;// óńňŕíîâëĺíčĺ ńâ˙çč ďđĺäűäóůĺăî óçëŕ ń íîâűě
		if (pkey != *pend)(pv->next)->prev = pv;// óńňŕíîâëĺíčĺ ńâ˙çč ďîńëĺäóţůĺăî óçëŕ ń íîâűě
		// Îáíîâëĺíčĺ óęŕçŕňĺë˙ íŕ ęîíĺö ńďčńęŕ,
		// ĺńëč óçĺë âńňŕâë˙ĺňń˙ â ęîíĺö
		else *pend = pv;
		return pv;
	}
	return 0;
}
